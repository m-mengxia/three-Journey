### 第一阶段：线性基础（1-10）

这部分主要理解 UV 坐标的单向变化。

1. **全黑/全白**：设置 `strength = 0.0` 或 `1.0`。
2. **水平渐变**：`vUv.x`（左黑右白）。
3. **垂直渐变**：`vUv.y`（下黑上白）。
4. **反向垂直渐变**：`1.0 - vUv.y`。
5. **急速压缩渐变**：`vUv.y * 10.0`。
6. **垂直循环渐变**：`mod(vUv.y * 10.0, 1.0)`。
7. **硬边垂直条纹**：`step(0.5, mod(vUv.y * 10.0, 1.0))`。
8. **更细的硬边条纹**：`step(0.8, mod(vUv.y * 10.0, 1.0))`。
9. **更粗的硬边条纹**：`step(0.2, mod(vUv.y * 10.0, 1.0))`。
10. **硬边水平条纹**：`step(0.8, mod(vUv.x * 10.0, 1.0))`。

```
varying vec2 vUv;

void main() {
    // 默认强度为 0
    float strength = 0.0;

    // --- 第一阶段：线性基础 (请取消注释或修改下面对应的逻辑) ---

    // 1. 全黑 / 全白
    // strength = 1.0; 

    // 2. 水平渐变：从左(0.0)到右(1.0)
    // strength = vUv.x;

    // 3. 垂直渐变：从下(0.0)到上(1.0)
    // strength = vUv.y;

    // 4. 反向垂直渐变：从下(1.0)到上(0.0)
    // strength = 1.0 - vUv.y;

    // 5. 急速压缩渐变：快速达到 1.0（白色区域更多）
    // strength = vUv.y * 10.0;

    // 6. 垂直循环渐变：利用 mod 产生 10 个 0->1 的循环
    // strength = mod(vUv.y * 10.0, 1.0);

    // 7. 硬边垂直条纹：使用 step 过滤，一半黑一半白
    // strength = step(0.5, mod(vUv.y * 10.0, 1.0));

    // 8. 细硬边垂直条纹：只有最后 20% 是白色
    // strength = step(0.8, mod(vUv.y * 10.0, 1.0));

    // 9. 粗硬边垂直条纹：80% 都是白色
    // strength = step(0.2, mod(vUv.y * 10.0, 1.0));

    // 10. 硬边水平条纹：将逻辑应用到 x 轴
    strength = step(0.8, mod(vUv.x * 10.0, 1.0));

    // 最终输出：将 strength 应用到 RGB，Alpha 保持 1.0
    gl_FragColor = vec4(vec3(strength), 1.0);
}
```

### 第二阶段：交叉与重叠（11-20）

开始利用 `+`（加法/并集）和 `*`（乘法/交集）构建网格。

11. **条纹十字架**：`step(0.8, mod(vUv.x * 10.0, 1.0)) + step(0.8, mod(vUv.y * 10.0, 1.0))`。
12. **虚线网格**：`step(0.8, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0, 1.0))`。
13. **长条虚线**：`step(0.4, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0, 1.0))`。
14. **L 型拐角**：结合上述逻辑，通过偏置实现两条线的交汇。
15. **T 型组合**：在水平和垂直方向做不同的 `step` 裁剪。
16. **中心距离（绝对值）**：`abs(vUv.x - 0.5)`（产生对称的中心渐变）。
17. **十字光芒**：`min(abs(vUv.x - 0.5), abs(vUv.y - 0.5))`。
18. **中心黑块**：`max(abs(vUv.x - 0.5), abs(vUv.y - 0.5))`。
19. **边框效果**：`step(0.2, max(abs(vUv.x - 0.5), abs(vUv.y - 0.5)))`。
20. **嵌套正方形**：利用 `floor` 或重复的 `max(abs)`。

```
varying vec2 vUv;

void main() {
    float strength = 0.0;

    // --- 11. 纵横条纹相加 (OR 逻辑) ---
    // 产生了格子边框，像一个十字架阵列
    // strength = step(0.8, mod(vUv.x * 10.0, 1.0)) + step(0.8, mod(vUv.y * 10.0, 1.0));

    // --- 12. 纵横条纹相乘 (AND 逻辑) ---
    // 只有交叉点会留下，产生了“波点/方块”阵列
    // strength = step(0.8, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0, 1.0));

    // --- 13. 长条虚线 (非对称相乘) ---
    // x 方向很宽 (0.4)，y 方向很窄 (0.8)，相乘后得到水平的长方块
    // strength = step(0.4, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0, 1.0));

    // --- 14. L 型拐角 (组合逻辑) ---
    // 通过两个条形组合出 L 形状（这个练习通常涉及多个 step 的加法）
    // float barX = step(0.4, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0, 1.0));
    // float barY = step(0.8, mod(vUv.x * 10.0, 1.0)) * step(0.4, mod(vUv.y * 10.0, 1.0));
    // strength = barX + barY;

    // --- 15. T 型/十字偏移 ---
    // 在 14 的基础上，通过给 vUv 加上偏移量来移动条纹位置
    // float barX2 = step(0.4, mod(vUv.x * 10.0, 1.0)) * step(0.8, mod(vUv.y * 10.0 + 0.2, 1.0));
    // float barY2 = step(0.8, mod(vUv.x * 10.0 + 0.2, 1.0)) * step(0.4, mod(vUv.y * 10.0, 1.0));
    // strength = barX2 + barY2;

    // --- 16. 中心距离（绝对值） ---
    // abs(vUv.x - 0.5) 把渐变变成了 V 字形，从中心向两边变白
    // strength = abs(vUv.x - 0.5);

    // --- 17. 十字光芒 (取最小值) ---
    // min 函数取两个轴最小的距离，形成了十字形的黑影渐变
    // strength = min(abs(vUv.x - 0.5), abs(vUv.y - 0.5));

    // --- 18. 中心黑方块 (取最大值) ---
    // max 函数取两个轴最大的距离，形成了一个从中心向四周扩散的正方形渐变
    // strength = max(abs(vUv.x - 0.5), abs(vUv.y - 0.5));

    // --- 19. 正方形边框 ---
    // 在 18 的基础上套用 step，过滤掉中心部分，只留下边缘
    // strength = step(0.2, max(abs(vUv.x - 0.5), abs(vUv.y - 0.5)));

    // --- 20. 嵌套正方形/画框 ---
    // 结合 18 和 19，通过两个正方形相减或取交集，做出一个细线框
    float square1 = step(0.2, max(abs(vUv.x - 0.5), abs(vUv.y - 0.5)));
    float square2 = 1.0 - step(0.25, max(abs(vUv.x - 0.5), abs(vUv.y - 0.5)));
    strength = square1 * square2;

    gl_FragColor = vec4(vec3(strength), 1.0);
}
```

### 第三阶段：数学函数进阶（21-30）

引入 `floor`（取整）实现像素化，`fract`（取小数部）实现循环。

21. **阶梯状像素**：`floor(vUv.x * 10.0) / 10.0`。
22. **棋盘格像素**：同时对 x 和 y 进行 `floor` 后相乘。
23. **随机噪点**：`fract(sin(dot(vUv, vec2(12.9898, 78.233))) * 43758.5453)`（经典的伪随机函数）。
24. **像素化随机**：结合 21 的 `floor` 和 23 的随机。
25. **带偏置的像素随机**：在随机函数中加入 UV 偏移。
26. **倾斜角度**：`length(vUv)`。
27. **中心圆点渐变**：`distance(vUv, vec2(0.5))`。
28. **反向日蚀**：`1.0 - distance(vUv, vec2(0.5))`。
29. **亮斑/镜头光晕**：`0.01 / distance(vUv, vec2(0.5))`。
30. **拉伸亮斑**：对 UV 进行不均匀缩放后的 `distance`。

```
varying vec2 vUv;

// 23. 经典伪随机函数 (需要定义在 main 之外)
// 原理：利用 sin 函数的高频振荡取其小数部分，产生看似无序的数值
float random(vec2 st) {
    return fract(sin(dot(st.vec2(12.9898, 78.233))) * 43758.5453123);
}

void main() {
    float strength = 0.0;

    // --- 21. 阶梯状像素 (floor) ---
    // floor 会向下取整。vUv.x * 10.0 的范围是 0-10，floor 后变成 0, 1, 2...9
    // 再除以 10.0，数值就变成了 0.0, 0.1, 0.2... 这种阶梯状
    // strength = floor(vUv.x * 10.0) / 10.0;

    // --- 22. 棋盘格像素 ---
    // x 和 y 同时阶梯化，产生方块阵列
    // strength = (floor(vUv.x * 10.0) / 10.0) * (floor(vUv.y * 10.0) / 10.0);

    // --- 23. 伪随机噪点 ---
    // 调用上面的 random 函数。每个像素都会得到一个基于坐标的随机值
    // strength = random(vUv);

    // --- 24. 像素化随机 ---
    // 先把 UV 坐标阶梯化，再传给随机函数。这样每个 0.1x0.1 的方块内随机值相同
    // vec2 gridUv = vec2(floor(vUv.x * 10.0) / 10.0, floor(vUv.y * 10.0) / 10.0);
    // strength = random(gridUv);

    // --- 25. 带偏置的像素随机 ---
    // 在 x 和 y 阶梯化时加入不同的偏移或比例，使方块变成长方形并产生错位感
    // vec2 gridUv2 = vec2(floor(vUv.x * 10.0) / 10.0, floor((vUv.y + vUv.x * 0.5) * 10.0) / 10.0);
    // strength = random(gridUv2);

    // --- 26. 长度渐变 (length) ---
    // 从左下角 (0,0) 开始计算向量长度。左下角最黑，右上角最白
    // strength = length(vUv);

    // --- 27. 中心圆点渐变 (distance) ---
    // 计算当前像素到中心点 (0.5, 0.5) 的距离
    // strength = distance(vUv, vec2(0.5));

    // --- 28. 反向中心渐变 (日蚀效果) ---
    // 1.0 减去距离，中心最亮，向外变暗
    // strength = 1.0 - distance(vUv, vec2(0.5));

    // --- 29. 亮斑/镜头光晕 (点源压缩) ---
    // 利用除法的特性：当距离接近 0 时，数值会爆炸增长。0.01 是为了控制光晕大小
    // strength = 0.01 / distance(vUv, vec2(0.5));

    // --- 30. 拉伸亮斑 (变形光晕) ---
    // 在计算距离前，对 vUv 的轴进行缩放。这里 y 轴被压缩，光晕会沿水平拉长
    vec2 lightUv = vec2(vUv.x * 0.1 + 0.45, vUv.y * 0.5 + 0.25);
    strength = 0.01 / distance(lightUv, vec2(0.5));

    gl_FragColor = vec4(vec3(strength), 1.0);
}
```

### 第四阶段：形状与极坐标（31-40）

31. **星星/十字星**：结合两个拉伸的亮斑（x 和 y 轴）。
32. **旋转中心**：通过矩阵旋转 UV 后的 31 练习。
33. **圆环**：`step(0.1, abs(distance(vUv, vec2(0.5)) - 0.25))`。
34. **波浪圆环**：在距离函数中加入 `sin`。
35. **锯齿圆环**：在距离函数中加入 `abs` 和 `sin`。
36. **极坐标角度**：`atan(vUv.x - 0.5, vUv.y - 0.5)`（旋转渐变的关键）。
37. **雷达/扇形渐变**：规范化后的 `atan`。
38. **螺旋线**：将 `atan` 与 `distance` 结合。
39. **多角星/花瓣**：利用 `atan` 产生的角度作为 `sin` 的频率输入。
40. **动态波纹**：在 `sin` 中引入变量（虽然这章主要是静态，但逻辑已通）。

```
varying vec2 vUv;

void main() {
    float strength = 0.0;

    // --- 31. 十字星光 (Star) ---
    // 结合两个拉伸的亮斑：一个水平拉伸，一个垂直拉伸，取乘法（交集）
    // float lightX = 0.015 / distance(vec2(vUv.x * 0.1 + 0.45, vUv.y * 0.5 + 0.25), vec2(0.5));
    // float lightY = 0.015 / distance(vec2(vUv.y * 0.1 + 0.45, vUv.x * 0.5 + 0.25), vec2(0.5));
    // strength = lightX * lightY;

    // --- 32. 旋转 45 度的十字星 ---
    // 在计算 31 之前，对 vUv 进行旋转矩阵变换
    // vec2 rotatedUv = rotate(vUv, PI * 0.25, vec2(0.5)); // 需要自定义 rotate 函数
    // strength = ... (同 31，但使用 rotatedUv)

    // --- 33. 圆环 (Circle Ring) ---
    // distance 产生圆，abs(distance - radius) 产生圆的轮廓，step 产生硬边
    // strength = step(0.2, 1.0 - abs(distance(vUv, vec2(0.5)) - 0.25));

    // --- 34. 波浪圆环 (Sine Ring) ---
    // 在 distance 中加入 sin 波形，使半径随角度或坐标摆动
    // strength = step(0.01, abs(distance(vUv, vec2(0.5)) - 0.25 + sin(vUv.x * 50.0) * 0.01));

    // --- 35. 锯齿圆环 ---
    // 使用 abs(sin) 产生重复的尖锐波峰
    // strength = step(0.01, abs(distance(vUv, vec2(0.5)) - 0.25 + abs(sin(vUv.x * 50.0)) * 0.05));

    // --- 36. 极坐标基础：atan ---
    // atan(y, x) 返回当前像素相对于中心点的角度（弧度制：-PI 到 PI）
    // strength = atan(vUv.x - 0.5, vUv.y - 0.5);

    // --- 37. 雷达扫描渐变 (Angle Normalized) ---
    // 将 atan 的范围映射到 0.0 到 1.0 之间
    // strength = (atan(vUv.x - 0.5, vUv.y - 0.5) / (PI * 2.0)) + 0.5;

    // --- 38. 螺旋线 (Spiral) ---
    // 将角度渐变与重复函数 mod 结合
    // float angle = (atan(vUv.x - 0.5, vUv.y - 0.5) / (PI * 2.0)) + 0.5;
    // strength = mod(angle * 20.0, 1.0);

    // --- 39. 花瓣/多角星 ---
    // 利用角度作为 sin 的频率，改变圆的半径
    // float angle = atan(vUv.x - 0.5, vUv.y - 0.5);
    // float wave = sin(angle * 10.0);
    // strength = step(0.01, abs(distance(vUv, vec2(0.5)) - (0.2 + wave * 0.02)));

    // --- 40. 终极形状：太阳/指纹 ---
    // 将 sin(angle) 映射到距离场，形成向外扩散的波纹
    float angle = atan(vUv.x - 0.5, vUv.y - 0.5);
    strength = sin(angle * 100.0);

    gl_FragColor = vec4(vec3(strength), 1.0);
}
```

### 第五阶段：噪声与高级纹理（41-50）

这一部分通常涉及引入外部的 **Perlin Noise** 或 **Simplex Noise** 函数。

41. **柏林噪声基础**：直接输出 `cnoise(vUv * 10.0)`。
42. **云朵纹理**：`step` 处理后的噪声。
43. **迷彩图案**：噪声叠加 `abs`。
44. **熔岩/地形图**：噪声的等高线表现。
45. **锐化噪声线条**：`1.0 - abs(cnoise(vUv * 10.0))`。
46. **噪声扰动 UV**：用噪声的值去修改 UV，再传给下一个函数（非常酷的流体感）。
47. **像素化噪声**：`floor` 与 `cnoise` 的结合。
48. **迷幻万花筒**：噪声与极坐标的终极碰撞。
49. **终极噪点混合**：多个频率噪声叠加。
50. **全彩噪声/彩色图案**：将不同的计算结果分别填入 `gl_FragColor` 的 R, G, B 通道。

```
varying vec2 vUv;

// --- 必须引入经典的 Perlin Noise 函数 (这里简写其逻辑) ---
float cnoise(vec2 P) {
    // ... 复杂的置换与插值算法 (从课程资源复制)
}

void main() {
    float strength = 0.0;

    // --- 41. 柏林噪声基础 (Basic Noise) ---
    // 产生像云雾一样的随机灰阶
    // strength = cnoise(vUv * 10.0);

    // --- 42. 噪点硬化 (Step Noise) ---
    // 产生像细胞、斑块或迷彩的效果
    // strength = step(0.0, cnoise(vUv * 10.0));

    // --- 43. 锐化噪声轮廓 (Abs Noise) ---
    // 利用 abs 把负值翻转，产生像电光、树根一样的细线
    // strength = 1.0 - abs(cnoise(vUv * 10.0));

    // --- 44. 熔岩/等高线 (Sin Noise) ---
    // 在噪声基础上套用 sin，产生像指纹或地形图的波纹
    // strength = sin(cnoise(vUv * 10.0) * 20.0);

    // --- 45. 锐化等高线 ---
    // 结合 43 和 44，让线条变细，产生极具科幻感的流动感
    // strength = step(0.9, sin(cnoise(vUv * 10.0) * 20.0));

    // --- 46. 域扭曲 (Domain Warping - 入门) ---
    // 这是最酷的技术：用一份噪声去修改另一份噪声的 UV
    // 会产生液体旋转、奶油搅拌的视觉效果
    // float distortion = cnoise(vUv * 10.0);
    // strength = cnoise(vUv * 10.0 + distortion);

    // --- 47. 像素化噪声 (Pixelated Noise) ---
    // 结合 floor 和 cnoise，产生复古游戏地图的效果
    // vec2 gridUv = floor(vUv * 30.0) / 30.0;
    // strength = cnoise(gridUv * 10.0);

    // --- 48. 噪点锐化边缘 (Smoothstep) ---
    // 使用 smoothstep 代替 step，获得带有羽化边缘的圆块
    // strength = smoothstep(0.0, 0.1, cnoise(vUv * 10.0));

    // --- 49. 迷幻万花筒 (Kaleidoscope Noise) ---
    // 将极坐标 (atan) 与噪声结合，噪声会围绕中心旋转
    // float angle = atan(vUv.x - 0.5, vUv.y - 0.5);
    // strength = cnoise(vec2(angle * 5.0, distance(vUv, vec2(0.5)) * 10.0));

    // --- 50. 炫彩输出 (Final Color Mix) ---
    // 将不同的计算逻辑分别填入 R, G, B 通道，或者利用 mix 函数在两种颜色间插值
    vec3 color1 = vec3(1.0, 0.0, 0.0); // 红色
    vec3 color2 = vec3(0.0, 0.0, 1.0); // 蓝色
    float mixer = step(0.0, cnoise(vUv * 10.0));
    vec3 finalColor = mix(color1, color2, mixer);

    gl_FragColor = vec4(finalColor, 1.0);
}
```
